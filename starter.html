Data Staging¶

We'll use a sample dataset called CIFAR to simulate the challenges Scones Unlimited are facing in Image Classification. In order to start working with CIFAR we'll need to:

  1. Extract the data from a hosting service
  2. Transform it into a usable shape and format
  3. Load it into a production system

In other words, we're going to do some simple ETL!

import os
def save_images(row,test_or_train):
    #Grab the image data in row-major form
    if test_or_train=='Train':
        dataset = dataset_train.copy()
        path = './train'
    elif test_or_train=='Test':
        dataset = dataset_test.copy()
        path = './test'
    else:
        raise ValueError("Invalid value for 'test_or_train'. Expected 'Train' or 'Test'.")
    
    img = dataset[b'data'][row['row']]
    
    # Consolidated stacking/reshaping from earlier
    target = np.dstack((
    img[0:1024].reshape(32,32),
    img[1024:2048].reshape(32,32),
    img[2048:].reshape(32,32)))
    
    # Save the image
    path = os.path.join(path, row['filenames'])
    plt.imsave(path, target)
    
    # Return any signal data you want for debugging
    return path

For Image Classification, Sagemaker also expects metadata e.g. in the form of TSV files with labels and filepaths. We can generate these using our Pandas DataFrames from earlier:

In [62]:
def to_metadata_file(df, prefix):
    df["s3_path"] = df["filenames"]
    df["labels"] = df["labels"].apply(lambda x: 0 if x==8 else 1)
    return df[["row", "labels", "s3_path"]].to_csv(
        f"{prefix}.lst", sep="\t", index=False, header=False
    )

def simple_getter(obj):
    # Decode the image
    b64_image = obj["captureData"]["endpointInput"]["data"]
    image_bytes = base64.b64decode(b64_image)
    image = Image.open(BytesIO(image_bytes))

    # Parse the inference JSON data
    output_data = obj["captureData"]["endpointOutput"]["data"]
    inference = json.loads(output_data)  # this gives a list of scores, e.g. [0.01, 0.99]

    return image, inference



# Plotting images with classification results
def plot_predictions(json_data, max_images=12):
    cols = 4
    rows = (min(len(json_data), max_images) + cols - 1) // cols
    plt.figure(figsize=(4 * cols, 4 * rows))

    for idx, obj in enumerate(json_data[:max_images]):
        image, inference = simple_getter(obj)
        max_score = max(inference)
        pred_class = "bicycle" if inference[0] > inference[1] else "motorcycle"

        plt.subplot(rows, cols, idx + 1)
        plt.imshow(image)
        plt.axis("off")
        plt.title(f"{pred_class}\nConf: {max_score:.2f}", fontsize=10,
                  color='green' if max_score > 0.94 else 'red')

    plt.suptitle("Model Predictions with Confidence", fontsize=16)
    plt.tight_layout()
    plt.subplots_adjust(top=0.9)
    plt.show()
